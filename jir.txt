function Initialize-LogFile {
    [CmdletBinding()]
    param (
        [string]$LogFilePrefix = "roleAssignment_status"  # Default prefix for log files
    )

    try {
        # Generate a timestamp for uniqueness
        $timestamp = Get-Date -Format "yyyyMMdd_HHmmss"

        # Create the log file name
        $logFileName = "$LogFilePrefix-$timestamp.log"

        # Create the file in the current directory
        $logFilePath = Join-Path -Path (Get-Location) -ChildPath $logFileName
        $logFile = New-Item -Path $logFilePath -ItemType File -Force -ErrorAction Stop

        # Write an initial log entry
        "Log file initialized at $($logFilePath)" | Out-File -FilePath $logFilePath -Append

        # Return the log file path
        return $logFilePath
    }
    catch {
        # If log file creation fails, throw a terminating error
        Write-Error "Failed to create log file in the current directory: $_"
        throw "Log file creation failed. Cannot continue."
    }
}
