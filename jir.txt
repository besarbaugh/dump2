# Check the value of $Operation to handle logic.
if ($Operation -eq 'create' -and !$PSBoundParameters['DurationInDays']) {
    throw "DurationInDays is required for the 'create' operation."
}

if ($Operation -eq 'delete' -and $PSBoundParameters['DurationInDays']) {
    throw "DurationInDays is not valid for the 'delete' operation."
}
